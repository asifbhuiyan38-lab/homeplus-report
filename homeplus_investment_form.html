function doPostInvestment(e) {
  var sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("Investment");

  if (!sheet) {
    sheet = SpreadsheetApp
      .getActiveSpreadsheet()
      .insertSheet("Investment");
    sheet.appendRow([
      "Timestamp","তারিখ","রিপোর্টকারী","সময়",
      "ST বিনিয়োগ","ST উত্তোলন","ST Profit","ST জন","ST নোট",
      "DIR বিনিয়োগ","DIR উত্তোলন","DIR Profit","DIR নোট",
      "SHARE বিনিয়োগ","SHARE উত্তোলন","SHARE Profit","SHARE নোট",
      "RES বিনিয়োগ","RES উত্তোলন","RES Profit","RES নোট",
      "মোট বিনিয়োগ","মোট উত্তোলন","মোট Profit",
      "সাধারণ নোট"
    ]);
  }

  var d = JSON.parse(e.postData.contents);

  var totalInvest   = d.st_invest + d.dir_invest + d.sh_invest + d.res_invest;
  var totalWithdraw = d.st_withdraw + d.dir_withdraw + d.sh_withdraw + d.res_withdraw;
  var totalProfit   = d.st_profit + d.dir_profit + d.sh_profit + d.res_profit;

  sheet.appendRow([
    new Date(), d.date, d.reporter, d.time,
    d.st_invest, d.st_withdraw, d.st_profit, d.st_count, d.st_note,
    d.dir_invest, d.dir_withdraw, d.dir_profit, d.dir_note,
    d.sh_invest, d.sh_withdraw, d.sh_profit, d.sh_note,
    d.res_invest, d.res_withdraw, d.res_profit, d.res_note,
    totalInvest, totalWithdraw, totalProfit,
    d.generalNote
  ]);

  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}

doPost function আপডেট করো
Code.gs file-এ যাও → পুরো code replace করো নিচেরটা দিয়ে:
function doPost(e) {
  var d = JSON.parse(e.postData.contents);

  // Investment form হলে আলাদা handler
  if (d.formType === "investment") {
    return doPostInvestment(e);
  }

  // Super Shop form (default)
  var sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("SuperShop");

  if (!sheet) {
    sheet = SpreadsheetApp
      .getActiveSpreadsheet()
      .insertSheet("SuperShop");
    sheet.appendRow([
      "Timestamp","তারিখ","শাখা","রিপোর্টকারী",
      "সময়","মোট বিক্রয়","নগদ",
      "DBBL","City","Pubali","One","UCB","মোট Card",
      "Nagad","bKash","QR","মোট MFS",
      "মোট হিসাব","পার্থক্য",
      "খরচ","খরচের ধরন","নোট"
    ]);
  }

  sheet.appendRow([
    new Date(), d.date, d.branch, d.reporter, d.time,
    d.totalSales, d.cash,
    d.dbbl, d.city, d.pubali, d.onebank, d.ucb, d.cardTotal,
    d.nagad, d.bkash, d.qr, d.mfsTotal,
    d.calcTotal, d.diff,
    d.expense, d.expenseType, d.notes
  ]);

  return ContentService
    .createTextOutput(JSON.stringify({status:"ok"}))
    .setMimeType(ContentService.MimeType.JSON);
}
